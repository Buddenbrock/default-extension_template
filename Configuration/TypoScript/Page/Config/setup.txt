# The PAGE CONFIG template.
#
# Only page.config settings are done here, like character sets, url and language.

config {
  # Administrator settings
  admPanel = {$config.adminPanel}
  debug = {$config.debug}

  doctype = html5
  htmlTag_setParams = class="no-js"

  # Character sets
  renderCharset = utf-8
  metaCharset = utf-8

  # Disable border attribute of img tags
  disableImgBorderAttr = 1

  # Disable Base Extension Wrap
  disableBaseWrap = 1

  # Cache settings
  sendCacheHeaders = 1
  enableContentLengthHeader = 1

  # Meaningful filenames in the typo3temp folder
  meaningfulTempFilePrefix = 50

  # URL Settings
  tx_realurl_enable = 1
  simulateStaticDocuments = 0

  # Link settings
  absRefPrefix = {$config.absRefPrefix}
  typolinkEnableLinksAcrossDomains = 1
  redirectOldLinksToNew = 1
  typolinkCheckRootline = 1
  uniqueLinkVars = 1

  # Remove targets from links
  intTarget =
  extTarget = _blank

  # Indexed Search
  index_enable = 1
  index_externals = 1

  # Code cleaning
  disablePrefixComment = 1

  # Move default CSS and JS to external file
  removeDefaultJS = external
  inlineStyle2TempFile = 1
  removeDefaultCss = 1

  # Protect mail addresses from spamming
  spamProtectEmailAddresses = 0

  # Comment in the <head> tag
  headerComment = Developed by: buddenbrock Media Productions - http://www.buddenbrock.nrw/

  # Disable <title> tag
  noPageTitle = 0
  pageTitleFirst = 1
  pageTitleSeparator = ::
  pageTitleSeparator.noTrimWrap = | | |

  # JS/CSS concatenation/compression
  concatenateJs = 0
  concatenateCss = 0
  compressJs = 0
  compressCss = 0
}
